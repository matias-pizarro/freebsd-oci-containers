**Step 2: Write a smoke test for versions.json structure**

This test validates the critical data file that drives the entire build system.

```python
"""Tests for versions.json structure and integrity."""

import json
from pathlib import Path

import pytest

REPO_ROOT = Path(__file__).resolve().parent.parent
VERSIONS_FILE = REPO_ROOT / "versions.json"


@pytest.fixture
def versions():
    """Load versions.json."""
    with open(VERSIONS_FILE) as f:
        return json.load(f)


def test_versions_file_exists():
    """Verify versions.json exists at the repository root."""
    assert VERSIONS_FILE.exists(), f"versions.json not found at {VERSIONS_FILE}"


def test_required_top_level_keys(versions):
    """Verify all required top-level keys are present."""
    required_keys = {
        "build_environment",
        "registries",
        "projects",
        "os_versions",
        "os_reference_versions",
        "os_security_patches",
        "image_digests",
    }
    assert required_keys.issubset(
        versions.keys()
    ), f"Missing keys: {required_keys - versions.keys()}"


def test_registries_have_required_fields(versions):
    """Verify each registry has required fields."""
    for name, registry in versions["registries"].items():
        assert "name" in registry, f"Registry '{name}' missing 'name'"
        assert "username" in registry, f"Registry '{name}' missing 'username'"


def test_projects_have_required_fields(versions):
    """Verify each project has required fields."""
    required_fields = {"name", "slug", "icon", "type", "versions", "context", "reference_project_version"}
    for name, project in versions["projects"].items():
        missing = required_fields - project.keys()
        assert not missing, f"Project '{name}' missing fields: {missing}"


def test_project_types_are_valid(versions):
    """Verify project types are one of the known values."""
    valid_types = {"os_variant", "upstream_adaptation"}
    for name, project in versions["projects"].items():
        assert project["type"] in valid_types, (
            f"Project '{name}' has invalid type '{project['type']}'"
        )


def test_os_reference_versions_are_consistent(versions):
    """Verify reference versions point to existing OS versions."""
    ref = versions["os_reference_versions"]
    assert "reference_os_major_version" in ref
    major = ref["reference_os_major_version"]
    assert major in versions["os_versions"], (
        f"Reference OS major version '{major}' not in os_versions"
    )
```
