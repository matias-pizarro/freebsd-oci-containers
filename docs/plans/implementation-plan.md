```python
"""Tests for versions.json schema validation."""

import json
from pathlib import Path

import pytest

VERSIONS_FILE = Path(__file__).resolve().parent.parent / "versions.json"


@pytest.fixture
def versions():
    """Load and return the parsed versions.json."""
    with open(VERSIONS_FILE) as f:
        return json.load(f)


class TestTopLevelStructure:
    """Verify versions.json has required top-level keys."""

    REQUIRED_KEYS = {
        "build_environment",
        "registries",
        "projects",
        "os_versions",
        "os_reference_versions",
    }

    def test_file_exists(self):
        assert VERSIONS_FILE.exists(), f"{VERSIONS_FILE} not found"

    def test_valid_json(self):
        with open(VERSIONS_FILE) as f:
            data = json.load(f)
        assert isinstance(data, dict)

    def test_required_keys_present(self, versions):
        missing = self.REQUIRED_KEYS - set(versions.keys())
        assert not missing, f"Missing top-level keys: {missing}"


class TestRegistries:
    """Verify registry configuration structure."""

    REQUIRED_FIELDS = {"name", "username"}

    def test_at_least_one_registry(self, versions):
        assert len(versions["registries"]) >= 1

    @pytest.mark.parametrize(
        "registry_id",
        ["github", "docker", "quay"],
    )
    def test_registry_has_required_fields(self, versions, registry_id):
        if registry_id not in versions["registries"]:
            pytest.skip(f"Registry {registry_id} not configured")
        registry = versions["registries"][registry_id]
        missing = self.REQUIRED_FIELDS - set(registry.keys())
        assert not missing, f"Registry {registry_id} missing: {missing}"


class TestProjects:
    """Verify project definitions."""

    REQUIRED_FIELDS = {"name", "slug", "type", "versions", "versioning_scheme"}
    VALID_TYPES = {"os_variant", "upstream_adaptation"}

    def test_at_least_one_project(self, versions):
        assert len(versions["projects"]) >= 1

    def test_project_has_required_fields(self, versions):
        for name, project in versions["projects"].items():
            missing = self.REQUIRED_FIELDS - set(project.keys())
            assert not missing, f"Project {name} missing: {missing}"

    def test_project_type_is_valid(self, versions):
        for name, project in versions["projects"].items():
            assert project["type"] in self.VALID_TYPES, (
                f"Project {name} has invalid type: {project['type']}"
            )

    def test_upstream_projects_have_upstream_urls(self, versions):
        for name, project in versions["projects"].items():
            if project["type"] == "upstream_adaptation":
                assert "upstreams" in project, (
                    f"Upstream project {name} missing 'upstreams' field"
                )

    def test_versions_is_nonempty_list(self, versions):
        for name, project in versions["projects"].items():
            assert isinstance(project["versions"], list), (
                f"Project {name}: versions must be a list"
            )
            assert len(project["versions"]) >= 1, (
                f"Project {name}: versions must not be empty"
            )


class TestOsVersions:
    """Verify OS version matrix structure."""

    def test_at_least_one_os_major(self, versions):
        assert len(versions["os_versions"]) >= 1

    def test_os_versions_have_minor_versions(self, versions):
        for major, minors in versions["os_versions"].items():
            assert isinstance(minors, dict), (
                f"OS {major}: minor versions must be a dict"
            )
            assert len(minors) >= 1, (
                f"OS {major}: must have at least one minor version"
            )
```

## Phase 4: Update Upstream Projects One by One

Phase 4 is a **repeatable template** applied to each image in priority order.
Each image gets its own feature branch and dedicated session(s). The template
below should be followed for each image.

### Image priority order

1. `base` (foundation — everything depends on this)
2. `zfs` (foundation — extends base with ZFS annotations)
3. `postgres` (highest demand, `ready_for_upstream_merge: true` in versions.json)
4. `python` (high demand, `ready_for_upstream_merge: true` in versions.json)
5. `node` (high demand)
6. `golang` (relatively straightforward upstream)
7. `nginx` (high demand)
8. `uv` (extends python)

## Cross-cutting Concerns

### Commit discipline

- At least one commit per logical step
- Conventional commit format (`feat:`, `fix:`, `chore:`, `refactor:`, `test:`, `docs:`, `ci:`)
- Never rebase
- Feature branches merged with `--no-ff`

### Testing discipline

- TDD: write failing tests first, then implement
- Every module has corresponding test file
- Integration marker for slow tests (image builds, network calls)
- CI runs all non-integration tests on every push

### Documentation discipline

- Every code change that affects user-facing behaviour gets a docs update
- Executable snippets in docs that double as smoke tests
- MkDocs site rebuilt and validated in CI
